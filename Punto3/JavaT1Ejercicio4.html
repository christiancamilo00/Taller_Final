4. Realizar un programa el cual solicite el nombre de una ciudad capital y el país. Al finalizar debe imprimir en pantalla el siguiente mensaje "La ciudad [Nombre de la Ciudad], es la capital del país [Nombre del País] 

                                                    
                                                package Taller1.CiudadPais;
                                                
                                                import java.util.Scanner;
                                                
                                                /**
                                                    *
                                                    * @author Christian
                                                    */
                                                public class CiudadPais {
                                                        public static void main(String args[]) {
                                                        String ciudad, pais; 
                                                        
                                                        Scanner sc = new Scanner(System.in);
                                                
                                                        System.out.println("Por favor escribir una ciudad capital ");
                                                        ciudad = sc.nextLine();
                                                        System.out.println("Por favor escribir el pais");
                                                        pais = sc.nextLine();
                                                                
                                                        System.out.println("La ciudad " + ciudad + " es la capital del pais " + pais );    
                                                
                                                        }
                                                }
